Java
$ sudo update-alternatives --config java
# add JAVA_HOME
$ sudo vim /etc/environment
add JAVA_HOME="/usr/lib/jvm/java-7-openjdk-amd64" # depends your jdk version
$ sudo vim ~/.bashrc
add two lines to the end of the file (before gvm if you install it already")
export JAVA_HOME="/usr/lib/jvm/java-7-openjdk-amd64"
export PATH=$JAVA_HOME/bin:$PATH

# 前后台程序切换
Ctrl + Z # stop current job, push to background
$ jobs # list the jobs in background
$ bg # push the current job to background
$ bg 1 # push the job 1 to backgound
$ fg # get the current job back to front
$ fg 1 ge the job 1 to front
Ctrl + C # close job

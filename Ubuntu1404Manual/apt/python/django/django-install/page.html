# install pip
$ sudo apt-get install python-pip

# install Django
$ sudo pip install Django==1.6

# creat mysite (assume djangoworkspace is the working folder)
$ cd djangoworkspace
$ django-admin.py startproject mysite

# 好像在1.4.0版本以后的mysite目录中有两层文件夹：/mysite:【/mysite:【init.py,settings.py,urls.py,wsgi.py】manage.py】
# Re: 放在里面一层的mysite里，你可以这样理解，外面一个mysite是一个工程，里面是一个站点，在站点里面可以创建应用。
# Run the server
$ python manage.py runserver [0.0.0.0:8080]

# set database TIME_ZONE in mysite/settings.py
# set database
$ python manage.py syncdb

# creat a new app
$ python manage.py startapp polls

# edit polls/models.py as https://docs.djangoproject.com/en/1.6/inro/tutorial01/

# edit mysite/settings.py
# add 'polls' into INSTALLED_APPS

# not real functional, just print out let you know what syncdb is gone to do
$ python manage.py sql polls

# create tables that don't exist in database
$ python manage.py syncdb

# invoke the Python shell
$ python manage.py shell

# if you want to use PostgreSQL
# install PostgreSQL (please do not install this database if you are not familiar with it)
$ sudo apt-get install python-psycopg2

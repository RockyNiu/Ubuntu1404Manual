# download phpwind
$ wget http://www.phpwind.com/downloads/forums/phpwind_v9.0_utf8.zip
$ unzip phpwid

# copy files to server root
$ sudo cp -r ./upload/ /var/www/phpwind

# change the default folder
$ cd /etc/apache2/sites-avilable
$ sudo cp 000-default.conf 000-default.conf.bak # backup
$ sudo vim 000-default.conf # change the DocumentRoot
$ sudo service apache2 restart
$ sudo chmod -R 777 /var/www/phpwind #do not recommend

# install windows azure linux agent
$ sudo apt-get install walinuxagent

# Ctrl+C
$ sudo waagent -force -deprovision

# clear history
$ export HISTSIZE=0

# open Windows Azure PowerShell
C:\> Get-AzurePublishSettingsFile

# log in the AzurePublish and save the setting file
C:\> Import-AzurePublishSettingsFile 'THE_SETTING_FILE_PATH'
C:\> Get-AzureSubscription

# Account -> DashBoard -> Container
# copy Container path as CONTAINER_PATH
C:\> Add-AzureVhd -Destination CONTAINER_PATH/ubuntu.vhd -LocalFilePath D:\Hyper-V\Ubuntu_phpwind.vhd

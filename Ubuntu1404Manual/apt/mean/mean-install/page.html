# NVM install
$ curl https://raw.githubusercontent.com/creationix/nvm/v0.6.1/install.sh | sh
# active nvm
$ source ~/.nvm/nvm.sh
$ vim ~/.bashrc # --> add the line "source ~/.nvm/nvm.sh" inside

# list remote version of nodejs
$ nvm ls-remote
# install NodeJS or using this way
$ nvm install v0.10.28

# install npm
$ sudo apt-get install npm

# install express
$ npm install express -g

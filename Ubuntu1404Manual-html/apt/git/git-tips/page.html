$ git init
$ git add .
[$ cat gitignore]
$ git commit -a -m "first commit" # -a 跳过git add 命令 # -m git message
$ git rm --cached readme.txt # 从工作区删除readme.txt文件
$ git rm log/\*.log # 从工作区中删除log/目录下的所有.log文件
$ git rm \*~ # 从工作区中删除当前目录以及子目录中所有以～结尾的文件 -f 物理删除 --cached 没有物理删除，设置为隐藏
$ git mv file_from file_to # 将文件file_from 改名为 file_to
$ git log # 查看提交历史
$ git log -p -2 #-p
 
选项展开显示每次提交的内容差异，用 
-2
 则仅显示最近的两次更新
［# more to be read for $ git log］...
$ git commit --amend # 修改最近一次提交
$ git diff README # 查看README文件变化
$ git show 8803 # show commit变化 （show后面输入commit id 前四位）
$ git checkout -b testing # same as $ git branch testing $ git checkout testing

# merge the branch testing to master and delete it
$ git checkout master
$ git merge testing
$ git branch -d testing


# 提交远程仓库
$ git remote add lu git@github.com:RockyNiu/Ubuntu1204Manual.git
$ git push lu master

# 得到远程仓库更新
$ git pull lu master

# fork other's repo
$ git clone others.git
$ cd others
$ git remote add upstream others.git
$ git fetch upstream
$ git gedit README
$ git add .
$ git commit -m "fork from others"
$ git push origin HEAD:master

# show ref
$ git show-ref

# rename
$ git remote rename lu lm
# delete
$ git remote rm

# .gitignore
$ sudo touch ~/.gitignore_global
$ gedit ~/.gitignore_global
$ git config --global core.excludesfile ~/.gitignore_global

# install maven2
$ sudo apt-get install maven2

# 下载代码 git clone http://git.oschina.net/shishuo/CMS.git

# 进入CMS目录
$ cd CMS

# 清理
$ mvn clean

# 编译
$ mvn compile

# 修改数据库配置文件
假设mysql 有用户rocky，密码abc123，那么
jdbc.user=rocky
jdbc.password=abc123
# 把 src/main/resources/shishuocms.properties 拷贝到 CMS目录下
# 修改shishuocms.properties里的超级管理员的email，和数据库连接的相关信息

# 用phpmyadmin新增一个mysql database cms

# 安装
$ mvn exec:java -Dexec.mainClass="com.shishuo.cms.Install"

# 运行
$ mvn jetty:run

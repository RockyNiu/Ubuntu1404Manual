Matlab-install
  • # download Matlab-2012b.iso to /home/Downloads
  • $ cd /home/Downloads
  • $ sudo mount -o loop Matlab-2012b.iso /mnt
  • $ cd /mnt
  • $ sudo ./install # I chose /home/rocky/Matlab/2012b/ as the root
  • # set the system path for matlab
  • $ gedit ~/.bashrc
  • # add a line at the end
  • export PATH=$PATH:/home/rocky/Matlab/2012b/bin
  • # add matlab shortcuton desktop:
  • $ gnome-desktop-item-edit --create-new ~/Desktop
  • # setting command: matlab -desktop

For Psychtoolbox

  • # install subversion
  • $ sudo apt-get install subversion
  • # install low-latency kernel
  • $ sudo add-apt-repository ppa:abogani/ppa
  • $ sudo apt-get uupdate
  • $ sudo apt-get install linux-lowlatency
  • # fix a few things for PTB to work
  • $ cd /home/rocky/Matlab/2012b/sys/os/glnxa64/
  • $ sudo mv libstdc++.so.6.0.13 old_libstdc++.so.6.0.13
  • $ sudo mv libstdc++.so.6 old_libstdc+.so.6
  • $ sudo apt-get install libdc1394-22
  • # fix a Matlab issue specific to Ubuntu 11.04(not sure for Ubuntu 12.04)
  • $ sudo ln -s /lib64/x86_64-linux-gnu/libc-2.13.so /lib64/libc.so.6
  • # add the user to the psychtoolbox group to get write permission of specific files
  • > PsychLinuxConfiguration
